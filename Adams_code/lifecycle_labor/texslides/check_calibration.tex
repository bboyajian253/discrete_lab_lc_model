	\begin{align*}
		V_j(k; a) = \max_{c,k',n} \quad &u(c, n) + \beta V_{j+1}(k'; a)\\
		s.t. \quad &(1+\tau_c)c+k'=Rk+\tau_0(wh_j(a)n)^{1-\tau_1}+T \\
	\end{align*}
\vspace{-8mm} \\
Notation
\bi \footnotesize
		\f $a$ is ability
		\f $k$ is stock of assets
		\f $c$ is consumption
		\f $n$ is labor (hours)
		\f $h_j(a)$ is human capital of type $a$ at age $j$ (full $h_j(a)$ known ex-ante)
		\f $V_j(k;a)$ is value of being in state $(j,k;a)$
		\bi \footnotesize
			\f retirement: for $j\geq J_r$, $n=0$
			\f final period: require $k'\geq 0$
		\ei 
\ei
